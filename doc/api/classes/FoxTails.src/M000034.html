  def self.get_standard_icons
    old_have_standard_icons = @have_standard_icons
    unless @have_standard_icons
      get_file_icons
      get_edit_icons
      get_browse_icons
      get_view_icons
      get_tree_icons
      get_misc_icons
      @have_standard_icons = true
    end
    old_have_standard_icons
  end
