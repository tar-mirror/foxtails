    def init_target(target = nil, state = :none)
      super target
      @state = state
      @state_setter = "#{state}=".intern
      @state_getter = "#{state}".intern
      @state_when   = "when_#{state}".intern
      @state_remove_observer = "remove_observer_#{state}".intern
    end
