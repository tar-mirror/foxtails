    def connect_to_target
      connect(SEL_COMMAND) do
        set_target_state(value)
      end

      connect(SEL_CHANGED) do
        set_target_state(value) if @dynamic
      end
      
      when_target_state CHANGES do |v|
        if v
          self.value = v
          enable
        else
          disable
        end
      end
      
      connect_to_target_list
    end
