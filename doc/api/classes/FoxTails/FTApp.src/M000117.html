    def get_icon name, *dirs
      for dir in dirs + @icon_path
        path = File.join(dir, name)
        begin
          data = File.open(path, "rb") {|f| f.read}
          case path
          when /png$/i
            icon = Fox::FXPNGIcon.new(self, data)
          when /gif$/i
            icon = Fox::FXGIFIcon.new(self, data)
          end
        rescue Errno::ENOENT
        else return icon
        end
      end
      nil
    end
