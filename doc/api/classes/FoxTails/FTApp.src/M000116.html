    def error(msg = "An error occurred.", dialog = false)
      if dialog or RUBY_PLATFORM =~ /win32/i
        window = getRoot()
        create unless @ft_app_created
        FXMessageBox.error(window, MBOX_OK, "Error", msg)
        exit(1)
      else
        Kernel.raise
      end
    end
