    def do_popup
      self.state = STATE_ENGAGED
      w = @make_popup.call
      w.create
      pop_x, pop_y = translateCoordinatesTo(getRoot, 0, height)
      w.popup(nil, pop_x, pop_y)
      w.grabKeyboard
      getApp().runModalWhileShown(w)
    ensure
      self.state = STATE_UP
    end
